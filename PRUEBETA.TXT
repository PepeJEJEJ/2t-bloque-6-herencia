# 🧱 **Clase Persona (superclase)**

```java
public class Persona {
    private String nombre;
    private String apellidos;
    private String dni;
    private String email;

    public Persona(String nombre, String apellidos, String dni, String email) {
        this.nombre = nombre;
        this.apellidos = apellidos;
        this.dni = dni;
        this.email = email;
    }

    public String getNombre() { return nombre; }
    public String getApellidos() { return apellidos; }
    public String getDni() { return dni; }
    public String getEmail() { return email; }

    public void setNombre(String nombre) { this.nombre = nombre; }
    public void setApellidos(String apellidos) { this.apellidos = apellidos; }
    public void setDni(String dni) { this.dni = dni; }
    public void setEmail(String email) { this.email = email; }

    @Override
    public String toString() {
        return "Nombre: " + nombre + " " + apellidos +
               ", DNI: " + dni +
               ", Email: " + email;
    }

    public String mostrarDetalles() {
        return nombre + " " + apellidos + " (" + dni + ")";
    }
}
```

# 🏢 **Clase Directivo**

```java
public class Directivo extends Persona {
    private String cargo;
    private int antiguedad;

    public Directivo(String nombre, String apellidos, String dni, String email,
                     String cargo, int antiguedad) {
        super(nombre, apellidos, dni, email);
        this.cargo = cargo;
        this.antiguedad = antiguedad;
    }

    public String getCargo() { return cargo; }
    public int getAntiguedad() { return antiguedad; }

    public void setCargo(String cargo) { this.cargo = cargo; }
    public void setAntiguedad(int antiguedad) { this.antiguedad = antiguedad; }

    public double pagaExtraAntiguedad() {
        return antiguedad * 25;
    }

    @Override
    public String mostrarDetalles() {
        return "El directivo " + getNombre() + " ocupa el cargo de " + cargo +
               " y tiene " + antiguedad + " años de antigüedad.";
    }

    @Override
    public String toString() {
        return super.toString() +
               ", Cargo: " + cargo +
               ", Antigüedad: " + antiguedad +
               ", Extra mensual: " + pagaExtraAntiguedad() + "€";
    }
}
```

# 🎓 **Clase Alumno**

```java
public class Alumno extends Persona {
    private String ciclo;
    private double notaMedia;

    public Alumno(String nombre, String apellidos, String dni, String email,
                  String ciclo, double notaMedia) {
        super(nombre, apellidos, dni, email);
        this.ciclo = ciclo;
        this.notaMedia = notaMedia;
    }

    public boolean estaAprobado() {
        return notaMedia >= 5;
    }

    @Override
    public String mostrarDetalles() {
        return "Alumno del ciclo " + ciclo +
               " con nota media " + notaMedia +
               " (" + (estaAprobado() ? "Aprobado" : "Suspenso") + ")";
    }

    @Override
    public String toString() {
        return super.toString() +
               ", Ciclo: " + ciclo +
               ", Nota media: " + notaMedia;
    }
}
```

# 🏫 **Clase CentroEducativo**

```java
public class CentroEducativo {

    private Directivo directivo;
    private Administrativo[] administrativos = new Administrativo[2];
    private Profesor[] profesores = new Profesor[5];
    private Alumno[] alumnos = new Alumno[10];

    private int contAdmin = 0;
    private int contProfe = 0;
    private int contAlumno = 0;

    public void setDirectivo(Directivo d) {
        if (directivo == null) {
            directivo = d;
        } else {
            System.out.println("Ya hay un directivo asignado.");
        }
    }

    public void addAdministrativo(Administrativo a) {
        if (contAdmin < administrativos.length) {
            administrativos[contAdmin++] = a;
        } else {
            System.out.println("No se pueden añadir más administrativos.");
        }
    }

    public void addProfesor(Profesor p) {
        if (contProfe < profesores.length) {
            profesores[contProfe++] = p;
        } else {
            System.out.println("No se pueden añadir más profesores.");
        }
    }

    public void addAlumno(Alumno a) {
        if (contAlumno < alumnos.length) {
            alumnos[contAlumno++] = a;
        } else {
            System.out.println("No se pueden añadir más alumnos.");
        }
    }

    public void mostrarCentro() {
        System.out.println("=== DIRECTIVO ===");
        System.out.println(directivo);

        System.out.println("\n=== ADMINISTRATIVOS ===");
        for (Administrativo a : administrativos) {
            if (a != null) System.out.println(a);
        }

        System.out.println("\n=== PROFESORES ===");
        for (Profesor p : profesores) {
            if (p != null) System.out.println(p);
        }

        System.out.println("\n=== ALUMNOS ===");
        for (Alumno al : alumnos) {
            if (al != null) System.out.println(al);
        }
    }
}
```
